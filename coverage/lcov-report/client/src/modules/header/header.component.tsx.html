/**
 * @class Header
 */
 
import * as React from 'react';
import Button from 'material-ui/Button';

import TopBar from '@sitegen/ui/src/components/top-bar';
import { Container, Environment, UserInfo } from './header.styles';

interface IUserInfo {
    icon?: string;
    name?: string;
}
 
export interface IHeaderProps {
    logo?: string;
    env: string;
    user: IUserInfo;
    logOutCallback: () => void;
}
 
export const Header = (props: IHeaderProps): JSX.Element => {
    const userName = props.user.name || 'username';
    const userIcon = props.user.icon || require('./assets/images/user.png');
 
    return (
        <TopBar>
            <Container>
                <Environment>Environment : {props.env}</Environment>
                <UserInfo icon={userIcon}>
                    <div className="user-info__icon" />
                    <div className="user-info__name">{userName}</div>
                    <div className="user-info__button">
                        <Button onClick={props.logOutCallback} color="primary" raised={true}>
                            Log out
                        </Button>
                    </div>
                </UserInfo>
            </Container>
        </TopBar>
    );
};
 
export default Header;
